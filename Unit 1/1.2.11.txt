public class Variables 
{
    public static void main(String[] args)
    {
        // 1. Declare and initialize a variable for each scenario:
        // A friend's name
        String friendName="Tracy the Turtle";
        // Your lucky number
        int number=4;
        // Whether or not you are you a morning person (the value 
        // should be either `true` or `false`)
        boolean morningPerson=true;
        // Your favorite snack
        String favoriteSnack="Goldfish";
        // The price of your favorite snack
        double price=4.99;
        // 2. Print out each variable:
        System.out.println(friendName);
        System.out.println(number);
        System.out.println(morningPerson);
        System.out.println(favoriteSnack);
        System.out.println(price);
        
        
    }
}
